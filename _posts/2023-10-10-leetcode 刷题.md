## 一、回溯

### 1. 解决什么问题

- 组合问题
- 排列问题
  - 强调顺序
- 切割问题
- 子集问题
- 棋盘问题
  - N 皇后
  - 解数独



### 2. 解决方法

- 回溯法都可以抽象成一个多叉树，
  - 树的大小：集合数量
  - 树的深度：递归深度
- 三部曲
  - 确定参数返回值
  - 确定终止条件
  - 确定单层搜索逻辑

- 递归 == 回溯
- 代码框架
  - 暴力遍历
  - 记录状态
  - 保存

- 多叉树遍历
  - 截止条件：到叶子结点

```c++
{
  if(终止条件){

    return;
  }
  
  for(集合元素){
      // 处理节点

    	// 回溯   =》 类似一个撤销的过程
  }
  
    //截止条件，将一个答案存入res
    if (ans_size == k) {
        res[row] = (int*)malloc(sizeof(int) * k);
        for (int i = 0; i < ans_size; i++) {
            res[row][i] = ans[i];
        }
        col[row++] = ans_size;
        return;
    }

    //跳过当前idx
    backtrack(nums, n, k, idx + 1);

    //选择当前idx
    ans[ans_size++] = nums[idx];//push操作
    backtrack(nums, n, k, idx + 1);
    ans_size--;//pop操作
}
```

```c++
class Solution {
public:
    vector<vector<int> > ans;
    void backtrack(int n, int k, int index, vector<int> path)
    {
        if(path.size() == k){
            ans.push_back(path);
            return;
        }
        
        for(int i=index; i<=n; i++)
        {
            path.push_back(i);
            backtrack(n, k, i+1, path);  // 包含此元素 // 递归
            path.pop_back();
        }
    }

        vector<int> path;
        backtrack(n, k, 1, path);
        return ans;
    }
```

```c++
class Solution {
public:
    vector<vector<string>> ans;
    vector<string> temp;
    int sum = 0;
    bool isCstring(string str)
    {
        int n = str.length();
        int l = 0, r=n-1;
        while(l<r)
        {
            if(str[l++] != str[r--]) return false;
        }
        return true;
    }

    vector<vector<string>> partition(string s) {
        int n = s.length();
        backtrack(s, n, 0, 0, sum);
        return ans;
    }

    void backtrack(string s, int n, int begin, int end, int sum)
    {
        
        // 是回文串
        if(sum == n){
            ans.push_back(temp);
            return;
        }

        if(end == n) return;
        for(int i=end; i<n; i++)
        {
            // 加入
            if( isCstring(s.substr(begin, i-begin+1)) ){     // 子串是回文串
                temp.push_back(s.substr(begin, i-begin+1));  // 切割
                // ans.push_back(temp);
                sum += (i-begin+1);                          // 总数量
                backtrack(s, n, i+1, i+1, sum); // 加入
                
                temp.pop_back();  // 不加入
                sum -= (i-begin+1);
            }
        }
        
    }
};
```

```c++
class Solution {
public:
    void solveSudoku(vector<vector<char>>& board) {
        // vector<vector<char>> temp = board;
        dfs(board, 0, 0);
    }

    bool is_valided(vector<vector<char>>& board, int row, int col, char item) {
        // 检查行和列
        for (int i = 0; i < 9; i++) {
            if (board[i][col] == item || board[row][i] == item) {
                return false;
            }
        }

        // 检查 3x3 方格
        int row_start = row - row % 3;
        int col_start = col - col % 3;
        for (int i = row_start; i < row_start + 3; i++) {
            for (int j = col_start; j < col_start + 3; j++) {
                if (board[i][j] == item) {
                    return false;
                }
            }
        }
        return true;
    }

    bool dfs(vector<vector<char>>& board, int row, int col)
    {
        if(row==9) return true;   // 单条路径

        if(col==9){ return dfs(board, row+1, 0);}
        
        if(board[row][col] != '.')  // 已经有
        {
            return dfs(board, row, col+1);  // 下一个
        }
        
        // 没有
        for (char c = '1'; c <= '9'; c++)
        {
            if(is_valided(board, row, col, c))
            {
                board[row][col] = c;  // yao
                if(dfs(board, row, col + 1)) {
                    return true;
                }
                board[row][col] = '.';      // bu yao
            }
        }
        return false;
    }
};
```

```c++
class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        // 单调栈
        stack<int> s;
        int n=temperatures.size();
        vector<int > res(n,0);
        s.push(0);
        for(int i=1; i<n ; i++)
        {
            if(temperatures[i] > temperatures[s.top()]){
                res[s.top()] = i - s.top();
                s.pop();
                while(s.size()!=0 && temperatures[i] > temperatures[s.top()]){
                    res[s.top()] = i - s.top();
                    s.pop();
                }
                s.push(i);
            }else{
                s.push(i);
            }
        }
        return res;
    }
};
```

```c++
class Solution {
public:
    int trap(vector<int>& height) {
        // 单调栈 首个 >= 的数值
        int ans = 0;
        int n = height.size();
        // vector<int> right(n, -1);
        stack<int> s;
        s.push(0);
        for(int i=1; i<n; i++)
        {
            if(height[i] > height[s.top()]){
                // right[s.top()] = i;
                int min_val = height[s.top()];
                s.pop();
                while(!s.empty()){
                    int max_val =  min(height[i], height[s.top()]);
                    int h = max_val - min_val;
                    ans += (h*(i- s.top()-1));
                    if(height[i]>=height[s.top()])
                        s.pop();
                    else break;
                    min_val = max_val;
                }
                s.push(i);
            }else{
                s.push(i);
            }
        }
        
        return ans;
    }
};
```

  - 物品
    - 重量（小）
    - 价值（大）
  - 背包
    - 能装的最大重量

- dp[j]：
  - 容量为j的背包所能装的最大价值为 dp[i]
  - 递推
    - 不放物品i：dp[j]
    - 放物品i：dp[j-weight[i]] + value[i]
    - dp[j] = max(dp[j], dp[j-weight[i]] + value[i])

```c++
            return 0;
        } 

        // 两种偷法 
        // 1. root，孙子节点
        int val = 0;
        if(root->left){
            if(root->left->left) val += rob(root->left->left);
            if(root->left->right)  val += rob(root->left->right);
        }
        if(root->right){
            if(root->right->left)  val += rob(root->right->left);
            if(root->right->right)  val += rob(root->right->right);
        }
      	// 2. 子节点
        dp[root] = max(root->val+val, rob(root->left) + rob(root->right));
        return dp[root];
    }
};
```

#### 买卖股票

- [121. 买卖股票的最佳时机](https://leetcode.cn/problems/best-time-to-buy-and-sell-stock/)

```c++
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        // 知道未来状态，在低谷买，在峰值卖
        int min_price = prices[0];
        int profit = 0;
        for(int i=1; i<prices.size(); i++)
        {
            profit = max(profit, prices[i]-min_price);
            min_price = min(min_price, prices[i]);
        }
        return profit;
    }
};
```

```c++
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        // 可以购买多次 涨价买
        int ans = 0;
        int n = prices.size();
        for(int i=1; i<n; i++)
        {
            int pro = prices[i]-prices[i-1];
            if(pro>0)
            {
                ans += pro;
            }
        }
        return ans;
    }
};
```
